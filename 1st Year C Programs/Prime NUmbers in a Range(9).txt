#include <stdio.h>

int main() {
    int start, end;


    printf("Enter the starting number ");
    scanf("%d", &start);
    printf("Enter the ending number ");
    scanf("%d", &end);

    

    for (int num = start; num <= end; num++) {
        int isPrime = 1;  

        if (num <= 1) {
            isPrime = 0;  
        } else {
         
            for (int i = 2; i < num; i++) {
                if (num % i == 0) {
                    isPrime = 0;  
                    break;
                }
            }
        }

      
        if (isPrime) {
            printf("%d ", num);
        }
    }

    printf("\n");
   
}
