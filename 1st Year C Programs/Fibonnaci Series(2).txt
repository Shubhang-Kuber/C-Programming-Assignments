#include<stdio.h>
#include<stdlib.h>
#include<math.h>
#include<string.h>

int  fib(int n){
    if(n<=0){
        return 0;
    }
    if(n==1){
        return 1;
    }
    return (fib(n-2)+fib(n-1));
}

int main(){
    int n,m;
     printf("Enter the number\n")   ;
     scanf("%i", &n);

      m = fib(n);
     printf("The Fibonnaci of the number is: %i ", m);
     return 0;

}
