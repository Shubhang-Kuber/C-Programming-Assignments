#include<stdio.h>
#include<stdlib.h>
#include<math.h>

int main(){
   int m;
   printf("Enter the order of the matrix");
   scanf("%i", &m);
   int arr1[m][m], arr2[m][m], arr3[m][m];

   printf("Enter the elements of the matrix1");
   for(int i=0; i<m; i++){
        for(int j=0; j<m; j++){
            scanf("%i", &arr1[i][j]);
        }
   }

   printf("Enter elements of matrix2");
   for(int i=0; i<m; i++){
    for(int j=0; j<m; j++){
        scanf("%i", &arr2[i][j]);
    }
}
   
for(int i=0; i<m; i++){
    for(int j=0; j<m; j++){
        arr3[i][j]= arr1[i][j]+arr2[i][j];
    }
}

for(int i=0; i<m; i++){
    for(int j=0; j<m; j++){
        printf("%d ", arr3[i][j]);
    }
    printf("\n");
}


   
}
