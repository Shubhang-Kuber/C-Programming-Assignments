#include<stdio.h>
#include<stdlib.h>
#include<math.h>

int main(){
   int m, count=0;
   printf("Enter the order of the matrix");
   scanf("%i", &m);
   int arr[m][m];

   printf("Enter the elements of the matrix");
   for(int i=0; i<m; i++){
        for(int j=0; j<m; j++){
            scanf("%i", &arr[i][j]);
        }
   }

   for(int i=0; i<m; i++){
    for(int j=0; j<m; j++){
        if(arr[i][j]!=0){
            count++;
        }
    }
}

printf("The count is: %d", count);


   
   
}
