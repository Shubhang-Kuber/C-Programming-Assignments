#include<stdio.h>
#include<stdlib.h>
#include<math.h>

int main(){
int n, result=0;
printf("Enter the number");
scanf("%i", &n);

int temp=n;

while(temp!=0){
int ld = temp%10;
result = result + ld*ld*ld;
temp=temp/10;
}
if(result==n){
printf("Armstrong Number");
}
else{
printf("Not Armstrong Number");
}

}
