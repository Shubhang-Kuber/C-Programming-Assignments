#include<stdio.h>
#include<stdlib.h>
#include<math.h>
#include<string.h>

int  fact(int n){
    if(n==0 || n==1){
        return 1;
    }
    return (n*fact(n-1));
}

int main(){
    int n,m;
     printf("Enter the number\n")   ;
     scanf("%i", &n);

      m = fact(n);
     printf("The factorial of the number is: %i5 ", m);
     return 0;

}
