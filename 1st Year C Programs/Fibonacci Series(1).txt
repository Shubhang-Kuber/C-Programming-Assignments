#include <stdio.h>

int main() {
    int n, first = 0, second = 1, next;
    printf("Enter the no of terms");
    scanf("%i", &n);

    printf("Fibonacci Series up to %d terms:\n", n);

    for (int i = 1; i <= n; i++) {
        printf("%d ", first);
        next = first + second;
        first = second;
        second = next;
    }

    
}
